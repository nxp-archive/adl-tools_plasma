 17: // the angle manager for the ecu

 19: // ---------------------- curr_angle ---------------
 20: int32 curr_angle(int32 time) 
 21: {
 22:   // returns best estimate of current angle using global angle, time and angle rate info
 23:   int32 d_angle, angle_now, dt;
 24:     
 25:   // find out what the angle was when the last tooth was seen
 26:   // then compute where we should be given that and the angle rate
 27:     
 28:   //printf("\ncurr_angle::last angle =%ld at %ld", measured_angle, measured_t);
 29:   dt = time - measured_t;
 30:   //printf("::dt=%ld", dt);
 31:   d_angle = dt * est_angle_rate;
 32:   angle_now = measured_angle + d_angle;
 33:   //printf("::dangle=%ld rate=%ld ==> angle now = %ld", d_angle, est_angle_rate, angle_now);
 34:   return angle_now;
 35: }
 36: 
 37: // ---------------------- angle_to_time ------------
 38: static int32 angle_to_time(int32 time, int32 target_angle) 
 39: {
 40:   // uses stored angle info to convert a future angle to expected time of arrival
 41:   // initially just use angle rate (wrap in accel later)
 42:   // returns how long you must wait to get to the angle
 43:         
 44:   int32 dt, angle_now;
 45:         
 46:   // estimate what our current engine angle is
 47:   angle_now = curr_angle(time);
 48:         
 49:   // see whether the target angle is more than this or less
 50:   // if less, then it is for the next time around
 51:         
 52:   /*
 53:     if (target_angle < angle_now) {
 54:     target_angle = target_angle + circle;
 55:     }
 56:   */    
 57:   // now compute how long we have to wait to get to target angle
 58:   dt = (target_angle - angle_now)/est_angle_rate; 
 59:         
 60:   //printf("..wait %ld ticks for angle", dt);
 61:   
 62:   if (dt < 0) {
 63:     printf("\n\t*** angle-to-time for %d at %d: negative dt %d",
 64:            target_angle, time, dt);
 65:     dt = 0;
 66:   }
 67:   return dt;
 68: }
 69: 
 70: // ---------------------- signal_cylinder ------------	
 71: void signal_cylinder(int32 wakeangle, Chan &awake) 
 72: {       
 73:   // compute the time we need to wake up at
 74:   int32 dt = angle_to_time(pTime(), wakeangle);
 75:   //printf("\n\tsignal::angle = %ld;dt -> %ld", wakeangle, dt);
 76:   
 77:   // tell the cylinder
 78:   awake.write(dt);  
 79: }

 81: // ---------------------- track ------------------	
 82: static void track(Chan &teeth, Chans &tooth_marks) 
 83: {
 84:   // a simple digital filter to extract tooth rate
 85:   // every time it gets a tooth, it broadcasts to the cylinders so they can do The Right Thing
 86:   int32 tooth, angle_rate1;
 87:         
 88:   pSetPriority(1);
 89:                             
 90:   measured_t = pTime();
 91: 
 92:   measured_angle = 0;
 93:   est_angle_rate = 0;
 94:   est_angle_accel = 0;
 95:   printf("\n%08lld::tooth digital filter starting..", pTime());
 96:         
 97:   // first sample
 98:   tooth = teeth.get();
 99:   measured_angle = tooth;
100:   est_angle_rate = tooth_chunk/(pTime() - measured_t);              // a tooth's worth of degrees in time dt
101:   measured_t = pTime();                                         // timestamp the data
102:         
103:   printf("\n%08lld::tooth digital filter received sample 1..", pTime());
104:   printf("\n\tangle = %d rate = %d, accel = %d ", measured_angle, est_angle_rate, est_angle_accel);
105:   Compute(50);
106:         
107:   // second sample
108:   tooth = teeth.get();
109:   angle_rate1 = tooth_chunk/(pTime() - measured_t);
110:   est_angle_rate = (est_angle_rate + angle_rate1)/2;
111:         
112:   measured_t = pTime();
113:   printf("\n%08lld::tooth digital filter received sample 2..", pTime());
114:   printf("\n\tangle = %d rate = %d", measured_angle, est_angle_rate);
115:         
116:   Compute(50);
117:         
118:   // loop
119:   
120:   while (true) {
121:     int32 dt;
122:             
123:     // tooth digital filter loop
124:     tooth = teeth.get();
125:             
126:     //load = ExecInfo(TaskInstructions);
127:             
128:     measured_angle = tooth * tooth_chunk;
129:             
130:     dt = pTime() - measured_t;
131:             
132:     //printf("\n%ld::tooth %ld seen -> angle %ld; measured time = %ld; dt = %ld", 
133:     //	Now, tooth, measured_angle, measured_t, dt);
134:     measured_t = pTime();
135:             
136:     if (dt > 0) {
137:       fprintf(logfile, "\nTDF::%8lld::tooth %d", pTime(), tooth);
138:                 
139:       est_angle_rate = tooth_chunk/dt;                              // rate from prev tooth
140:       //printf(" ==> est_angle_rate = %ld", est_angle_rate); 
141:                 
142:     }
143:                 
144:     // now notify all cylinders of tooth arrival
145:     pfor (int i = 0; i != num_cylinders; ++i) {
146:       tooth_marks[i].write(tooth);
147:     }
148:             
149:     //load = ExecInfo(TaskInstructions) - load;
150:     //printf("\n---tooth digital filter loop - compute() is for %ld ticks", load);
151:     Compute(350);
152:   } 
153: }
154: 
155: // --------------------- cylinder_wake -----------------    
156: static void cylinder_wake(int32 index, Chan &tooth, Chan &ask, Chan &awake) 
157: {
158:   // one instance per cylinder
159:   // tooth counts toothticks from the digital filter; 
160:   // ask specifies a tooth to wait for from my cylinder
161:   // awake sigals back to the cylinder, the value being the time at which he should awake
162:         
163:   //int32 waketime;                 // the time to wake up at
164:   int32 waketooth;          // the tooth to wake up on
165:   int32 currtooth;          // the tooth we just saw
166:   int32 wakeangle;          // the angle we want to wake up at
167:         
168:   printf("\nCylinder-wake[%d] starting...", index);
169:         
170:   waketooth = -1;
171:   currtooth = -2;
172:   wakeangle = 0;
173:         
174:   pSetPriority(1);
175:         
176:   while (true) {
177:     alt {
178:       tooth.port(int ct) {
179:         currtooth = ct;
180: 
181:         // we just got a current tooth
182:         //printf("\n\tcylinder-wake[%ld] saw tooth %ld want t %ld", index, currtooth, waketooth);
183:         if (waketooth == currtooth) {
184:           // note the match
185:           //printf("\n\tcylinder-wake[%ld] saw tooth %ld matches tooth %ld for wakeangle %ld", 
186:           //    index, currtooth, waketooth, wakeangle);
187:     
188:           signal_cylinder(wakeangle, awake);
189:           waketooth = -1;
190:           Compute(83);
191:         }
192:       }
193:                     
194:       ask.port(int wa) {
195:         wakeangle = wa;
196:         int32 angle_now;
197:         int32 target_tooth;
198:         // we just got given an angle request, which we've captured in wakeangle                    
199:                     
200:         // see if this is for a smaller angle than where we're already at
201:         angle_now = curr_angle(pTime());
202:         target_tooth =      wakeangle/tooth_chunk;
203:         //printf("\n%ld::cylinder-wake[%ld] saw request for %ld; curr angle ~ %ld waketooth=%ld target_tooth=%ld", 
204:         //	Now, index, wakeangle, angle_now, waketooth, target_tooth); 
205:         if (angle_now > wakeangle) {
206:           // then we need to wake up in the future CHECK THIS FIRST!!
207:           waketooth = wakeangle / tooth_chunk;              // tooth before the angle we want
208:           //printf("==> toothchunk = %ld tooth wanted = %ld", tooth_chunk, waketooth);
209:           Compute(77);
210:         } 
211:         else if (target_tooth == currtooth) {
212:           // ok, we're still in this tooth
213:           //printf("==> got request for currtooth %ld == targettooth", currtooth);	
214:           waketooth = -1;
215:           signal_cylinder(wakeangle, awake);
216:           Compute(95);
217:         }
218:         else {
219:           waketooth = wakeangle / tooth_chunk;
220:           //printf("==> need new waketooth %ld", waketooth);
221:           Compute(51);
222:         }
223:       }
224:     }
225:   }
226: }
227:     
228: // ---------------------- angle_manager -----------------
229: void angle_manager(int32 num_cylinders, Chan &teeth, Chans &ask, Chans &wakeups) 
230: {
231:   // we get raw ticks on teeth (we get the tooth number)
232:   // ask[] provides a request that a client gets woken up at a specific angle
233:   // clients being numbered by index
234:   // client suspends on responses[index] and gets woken up when angle turns up
235:   
236:   Chans tooth_marks(cylinder_max);
237:   Chans requests(cylinder_max);
238:   Chan tooth_request;
239:                                                             
240:   pSetPriority(1);
241:         
242:   par {
243:     pfor(int i = 0; i != num_cylinders; ++i) {
244:       //service_request(i, ask[i], tooth_request, tooth_marks[i], wakeups[i]);
245:       cylinder_wake(i, tooth_marks[i], ask[i], wakeups[i]);
246:     }
247:     track(teeth, tooth_marks);
248:   }
249: }

  1: //
  2: // Copyright (C) 2005 by Freescale Semiconductor Inc.  All rights reserved.
  3: //
  4: // You may distribute under the terms of the Artistic License, as specified in
  5: // the COPYING file.
  6: //
  7: //
  8: // engine.pa
  9: //
 10: 
 11: #include <cstdio>
 12: #include <cmath>
 13: 
 14: #include "configuration.h"
 15: 
 16: using namespace std;
 17: using namespace plasma;
 18: 
 19: // ----------------------- accelerate --------------------
 20: 
 21: static double accelerate(double t, double fuel, double fuelpush, 
 22:                          double mass, double u) 
 23: {
 24:   // t is time interval                         : seconds
 25:   // fuel is amount of fuel burned in dt        : ml
 26:   // fuelpush is 'specific impulse' of fuel     : newton-seconds per ml
 27:   // mass is vehicle mass                       : kg
 28:   // u is initial speed                         : m/sec
 29:   // returns new speed                      : m/sec
 30:     
 31:   // Physics: Newtonian acceleration
 32:   // v**2 = u**2 + 2as
 33:   // s = ut + 0.5 * a * t**2
 34:   // p = ma;
 35:   // a = m/((fuel * fuelpush) - frictional_force)
 36:   // Friction: rises sharply with speed
 37:   // f = (friction * v) ** 2
 38:     
 39:   double s, v, a, f;
 40:     
 41:   //printf("\naccelerate: t = %f, fuel = %f, fuelpush = %f, mass = %fm u = %f",
 42:   //    t, fuel, fuelpush, mass, u);
 43:         
 44:   // compute effects of friction
 45:   f = friction * u;                         // assume speed constant enough over interval, and..
 46:   f = f * f ;                               // .. compute frictional force just for starting speed
 47:   //printf("\n\tfrictional force = %f", f);
 48:         
 49:   f = (fuel * fuelpush) - f;                // force = fuel's impulse - frictional force
 50:     
 51:     
 52:   //printf("effective force = %f", f);
 53:   a = f / mass;                             // p = ma, rearranged
 54:     
 55:   //printf("\n\ta = %f", a);
 56:     
 57:   s = (u * t) + ((a * t * t)/2.0);      // how far have we gone?
 58:   //printf("\n\ts = %f", s);
 59:     
 60:   v = (u * u) + (2 * a * s);                // compute new speed
 61:   //printf("\n\tvsqu = %f", v);
 62:     
 63:   // if v is negative, somehow friction is driving our car backwards, which is unlikely
 64:   // thus our dynamic simplifications must be at fault. just make car motionless
 65:   int32 v_int = (int32)v;
 66:   if (v_int < 0) {
 67:     // ooo! strong friction done slowed us to a halt
 68:     v = 0.0;
 69:   } else {
 70:     // otherwise compute sqrt
 71:     v = sqrt(v); 
 72:   }
 73:   
 74:   //printf("\n\tv = %f", v);

178: // ----------------------- flywheel ----------------------
179: void flywheel(Chan &from_flywheel, Chan &to_starter) 
180: {
181:   // models the flywheel's generation of flywheel ticks
182:   // we simply sample car speed at some fast enough rate
183:   // and compute when we'll see the next tick from a flywheel tooth
184:   // output a tick down the channel when it's time
185:   // we convert speed to degrees per second using speed ratio via rpm
186:   // we convert to teeth per second and look for the moments when elapsed
187:   // teeth equals next tooth 
188:     
189:   int32 engine_running = 1;
190:   double teeth = 0.0;                   // floaty representation of where we are in the teeth
191:   int32 next_tooth = 1;                 // integer number of next tooth
192:   bool started = false;
193:             
194:   printf("\nFlywheel starting...");
195:     
196:   while (true) {
197:     // sample car speed and compute when next tooth will hit
198:     double fraction = 0.02;                     // expect 50 samples per tooth interval
199:     double dt;                              // time interval
200:     double new_speed;                       // intermediate convenience value
201:     int32 wait_time;
202:     int32 sampling;             
203:     double tooth_target;
204:     double old_speed;
205: 
206:     sampling = true;
207:     //printf("\n%ld::\tFlywheel", Now);
208:     //teethpersec = tps(speed);
209:     //printf("\nSpeed = %f; teethpersec = %f", speed, teethpersec);
210:         
211:     dt = (next_tooth - teeth) / tps(speed);     // 100% of time to next tooth at curr speed
212:     dt = dt * fraction;                         // fraction of expected time
213:     wait_time = (int32) dt;
214:     if (wait_time == 0) {
215:       wait_time = 1;
216:     }
217:     tooth_target = next_tooth;
218:     old_speed = speed;
219:         
220:     //printf("\n%ld::Flywheel-compute delay till tooth %ld waittime=%ld", 
221:     //	Now, next_tooth, wait_time);
222:     //printf("speed=%2.2f teeth = %2.2f dt=%2.2f", speed, teeth, dt);
223:     while (sampling) {
224:       double dtooth;
225:       double teethpersec;
226:             
227:       pDelay(wait_time);
228:             
229:       // where have we got to with teeth?
230:       teethpersec = tps(speed);             // recompute because speed can change twixt teeth
231:       teeth = teeth + (teethpersec * dt);
232:       dtooth = tooth_target - teeth;
233:       if (dtooth <= 0.0) {
234:         // got to tooth time
235:         sampling = false;
236:       }
237:     }
238:     //printf("\n\n ==== ToothTick! %f", teeth);
239:         
240:     // on every tooth, compute car speed anew
241:     new_speed = accelerate(1.0, 0.0, 0.0, mass, speed);
242:     //fprintf(logfile, "\n\tfriction is slowing car: u=%2.2f v=%2.2f", speed, new_speed);
243:         
244:     fprintf(logfile, "\nTFS::%8lld::Tooth ", pTime(), next_tooth);
245:     fprintf(logfile, "speed %2.2f -- %2.2f", speed, new_speed);
246:     speed = new_speed;
247:         
248:     // create tooth interrupt
249:     from_flywheel.write(next_tooth);
250:         
251:     // set up for next tooth
252:     //printf("\n%ld::Created tooth interrupt t[%ld]", Now, next_tooth);
253:         
254:     teeth = next_tooth;
255:     next_tooth = next_tooth + 1;
256:     //printf("\t==>teeth = %2.2f", teeth);
257:         
258:     if (next_tooth == num_teeth) {
259:       // handle cyclic nature of tooth number
260:       next_tooth = 0;   
261:       teeth = 0.0;
262:       engine_running = engine_running - 1;
263:       
264:       // handle starter
265:       if (!started) {
266:         started = true;                 // yay! we're running!
267:         to_starter.write(true);
268:       }
269:     }
270:   } 
271: }

 20: /* ========================= description =======================

 22: the goal here is to get the 'architecture' of an engine controller captured, so that
 23: we can understand how the bits play together. when complete, this model will NOT contain (eg) 'real
 24: strategy code', but will contain at least placeholders for real code which could be refined
 25: in a later model

 27: what it will do is to allow us to 
 28:     - look at context switch rates
 29:     - look at partitioning the sw across more than one processor
 30:     - look for APU opportunities
 31:     - confirm the architecture
 32:     - look at the gross effects of scheduling and multiple processors on event jitter
 33: 
 34: for a next step:
 35: with 'real code' and a 'real OS' we can construct an exact analogue of this system at the 
 36: 'algorithmic' level of deatil, in which plausible code is running in a simulated realtime 
 37: environment. then we can further
 38:     - look at the effects of task switching on cache behaviour
 39:     - look for mechanisms which speed actual code
 40:     - look for mechanisms which shrink code footprint
 41:     - confirm it's code that needs shrining, not data
 42:     - compare 'locks' and 'messages' for footprint and efficiency
 43:     - ...
 44:     
 45: 
 46: what we model is a (very!) simplified engine and engine controller
 47:     - the engine is an n-cyclinder engine
 48:     - it has a flywheel with t teeth on it, sensed by a hall-effect sensor whose sensitivity
 49: varies with speed (low at low speeds) and which needs thresholding for effective digitisation
 50:     - there is an accelerator pedal
 51:     - there is an external environment which imposes frictional and mass-based forces on the motor
 52: so that there is a maximum speed it can attain with any particular rate of fuel feed and a 
 53: maximum acceleration at a given fuel feed rate
 54:     - there is a fuel injector per cylinder, which squirts fuel in when the injector is open. the
 55: amount of fuel injected is a function of fuel line pressure and open time
 56:     - there is a spark plug per cylinder, which must be zapped to have the fuel-air mixture go bang. 
 57: however, if it's asked to go bang at the wrong time, it might knock - get pre-ignition - nasty
 58:     - there is a fuel pressure detector
 59:     - there is a microphone per cylinder to listen for knocking
 60:     
 61: we will eventually try partitioning this software across a different mix of cpus, including a
 62: zen, an etpu and a cpm (or rather, logical equivalents thereof).

140: // ---------------------- cylinder_process ---------------
141: void cylinder_process(int32 c, int32 n, FuelChan &inject, Chan &spark, Chan &ask, Chan &wakeup) 
142: {
143:   // this process controls a single cylinder c of n cylinders, choosing how much
144:   // fuel to squirt, when, and when to fire
145:   // the firing order is an (unrealistic) 0, 1, 2, 3....
146:   // inject is the channel we tell cylinder to inject fuel, value is amount
147:   // spark is channel we tell cylinder's sparkplug to fire
148:   // ask is the channel we ask to be woken up at the appropriate angle
149:     
150:   // first choose our (fixed) firing angle (get adaptive later)
151:   // cylinder 0 fires at nominal angle 0.1, cylinder 1 at 0.1 + (1/n) later 
152:   //...angles measured as fractions of a circle
153:   // fuel needs to be squirted before ignition
154:   // since angular velocity of engine changes, we can't predict the time at
155:   // which the angle is reached massively in advance (although
156:   // we'll try by having a 2nd-order fit filter on angular velocity of flywheel
157:     
158:   int32 fire_angle, inject_angle;
159:     
160:   printf("\n\tCylinderManager Process starting for cylinder[%d] on ", c);
161:     
162:   inject_angle = c * (circle/(n + 1));          // compute default fire angle for cylinder c of n+1
163:   inject_angle = inject_angle + (circle/100);
164:   fire_angle = inject_angle + (circle/200);
165:     
166:   printf("\n\t..injects at %d, fires at %d", inject_angle, fire_angle);
167:     
168:   // now, loop
169:     
170:   while (true) {
171:     int32 accelerator, dt;
172:     int32 req_angle;
173:     fuel_request f_req;
174:     // find out where the accelerator is
175:         
176:     accelerator = gAccelerator;
177:     //printf("\n\t***c[%ld] reads accelerator as %ld\n", c, accelerator);
178:         
179:     // placeholder for strategy code to compute where we need to be
180:         
181:     Compute(3500);
182:         
183:     //printf("\n\t***c[%ld] has completed strategy code", c);
184:         
185:     // set up for injection
186:         
187:     //printf("\n\t***c[%ld] wants to wake up for angle %ld", c, inject_angle);
188:         
189:     ask.write(inject_angle);            // ask to be told when we get to the tooth before this angle
190:         
191:     dt = wakeup.get();              // get given time we need to suspend for        
192:     pDelay(dt);
193:         
194:     req_angle = curr_angle(pTime()) - inject_angle;
195:     num_samples = num_samples + 1;
196:     deadline_average[c] = deadline_average[c] + req_angle;
197:     deadline_variance[c] = deadline_variance[c] + (req_angle * req_angle);
198:     // tell injector to inject
199:     f_req.squirt_time = 1;
200:     f_req.fuel_amount = accelerator * 3;
201:         
202:     //printf("\n\t***c[%ld] injecting at %ld; ", c, Now);
203:     //printf("fuel_amount = %2.2f", f_req.fuel_amount);
204:     inject.write(f_req);
205: 
206:     // set up for fire
207:     printf("\n%lld::***c[%d] INJECTED (dtheta=%d)", pTime(), c, req_angle);
208:         
209:     ask.write(fire_angle);
210:     dt = wakeup.get();
211:     //printf("\n%ld::c[%ld] told to sleep for %ld ticks to reach fire angle", Now, c, dt);
212:         
213:     // sleep till we get here
214:     pDelay(dt);
215:         
216:     //printf("\n\t***c[%ld] firing at %ld", c, Now);
217:     req_angle = curr_angle(pTime()) - fire_angle;
218:     num_samples = num_samples + 1;
219:     deadline_average[c] = deadline_average[c] + req_angle;
220:     deadline_variance[c] = deadline_variance[c] + (req_angle * req_angle);
