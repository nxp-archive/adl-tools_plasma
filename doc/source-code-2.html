 21: // The architecture- global for now.

 24: // A tick of 10 units

 30: // Stage identifiers.
 31: const int FetchStage = 0;
 32: const int DecodeStage = 1;
 33: const int ExecStage = 2;
 34: const int MemStage = 3;
 35: const int WritebackStage = 4;
 36: 
 37: // Instruction classes.
 38: const int Load   = 0x0001;
 39: const int Store  = 0x0002;
 40: const int Branch = 0x0004;
 41: 
 42: // This delays by one clock cycle.
 43: void atclock(int n = 1)
 44: {
 45:   pDelay(Clock * n);
 46: }
 47: 
 48: // This creates a pipeline bubble by
 49: // stalling a stage for a clock cycle.
 50: void stall(int n,ClkInstChan &chan)
 51: {
 52:   for (int i = 0; i != n; ++i) {
 53:     chan.write(instr_t());
 54:     pDelay(Clock);
 55:   }
 56: }
 57: 
 58: //
 59: // Start of micro-architecture description.
 60: //
 61:         
 62: void fetch(NbInstChan &succ_chan,ClkInstChan &branch_update, 
 63:            ClkIntChan &refetch_chan, ClkInstChan &out)
 64: {
 65:   while (true) {
 66:     atclock();
 67: 
 68:     instr_t bupdate = (branch_update.ready()) ? branch_update.get() : 0;
 69:     instr_t instr = arch.fetch(bupdate);
 70:     update_pipe_report(FetchStage,instr);
 71:     instr_t succ = succ_chan.get();
 72: 
 73:     int refetch = (refetch_chan.ready()) ? refetch_chan.get() : 0;
 74:     if (succ.is_of_class(Branch)) {
 75:       refetch = true;
 76:     }
 77: 
 78:     succ_chan.write(instr); 
 79:     if (instr.is_of_class(Branch) || refetch) {
 80:       // This is just needed to keep us from fetching past a branch
 81:       // until we know the target.  This mainly just matters to avoid
 82:       // fetching past a halt.
 83:       arch.incr_iar(-4);
 84:     }      
 85:     if (refetch) {
 86:       succ_chan.write(instr_t()); 
 87:       continue;
 88:     }
 89: 
 90:     out.write(instr);
 91:   }
 92: }
 93: 
 94: void decode(ClkInstChan &in, ClkIntChan &refetch_chan, ClkInstChan &branch_update,
 95:             NbInstChan &succ_chan, NbInstChan &exec_fwd, NbInstChan &mem_fwd, 
 96:             ClkInstChan &out)
 97: {
 98:   while (true) {
 99:     instr_t instr = in.get();
100:     update_pipe_report(DecodeStage,instr);
101: 
102:     while (true) {
103:       arch.read_operands(instr);
104:       instr_t succ  = succ_chan.get();
105:       instr_t exec  = exec_fwd.get();
106:       instr_t mem   = mem_fwd.get();
107: 
108:       // Check for stalls here:
109:       //  Prior instruction is a load and our op uses the target.
110:       //  This instruction is a branch and op uses prior target.
111:       if (succ.is_of_class(Load) && arch.operand_conflict(instr,succ)) {
112:         // Insert bubble.
113:         stall(1,out);
114:         // Issue instruction.
115:         succ_chan.write(instr);
116:         out.write(instr);
117:         // Go to next cycle.
118:         atclock();
119:         break;
120:       } else if (instr.is_of_class(Branch) ) {
121:         if (arch.operand_conflict(instr,succ)) {
122:           refetch_chan.write(1);
123:           succ_chan.write(instr_t());
124:           stall(1,out);
125:           continue;
126:         }
127:       }
128:       // Branches update the iar in decode.
129:       if (instr.is_of_class(Branch)) {
130:         arch.update_from_fwd(instr,exec);
131:         arch.update_from_fwd(instr,mem);
132:         arch.exec(instr);
133:         branch_update.write(instr);
134:       }
135:       // Normal condition- pass data to next stage.
136:       succ_chan.write(instr);
137:       out.write(instr);
138:       break;
139:     }
140:   }
141: }
142: 
143: void exec(ClkInstChan &in, NbInstChan &mem_fwd_chan, 
144:           NbInstChan &exec_fwd_chan, NbInstChan &decode_fwd_chan, 
145:           ClkInstChan &out)
146: {
147:   while (true) {
148:     instr_t instr = in.get();
149:     update_pipe_report(ExecStage,instr);
150:     instr_t mem_fwd = mem_fwd_chan.get();
151:     instr_t exec_fwd = exec_fwd_chan.get();
152:     arch.update_from_fwd(instr,exec_fwd);
153:     arch.update_from_fwd(instr,mem_fwd);
154:     if (!instr.is_of_class(Load | Store)) {
155:       arch.exec(instr);
156:     }
157:     out.write(instr);
158:     exec_fwd_chan.write(instr);
159:     decode_fwd_chan.write(instr);
160:   }
161: }
162: 
163: void mem(ClkInstChan &in,NbInstChan &decode_fwd_chan,NbInstChan &exec_fwd_chan,
164:          NbInstChan &mem_fwd_chan,ClkInstChan &out) 
165: {
166:   while (true) {
167:     instr_t instr = in.get();
168:     update_pipe_report(MemStage,instr);
169:     instr_t mem_fwd = mem_fwd_chan.get();
170:     arch.update_from_fwd(instr,mem_fwd);
171:     arch.exec(instr);
172:     out.write(instr);
173:     decode_fwd_chan.write(instr);
174:     exec_fwd_chan.write(instr);
175:     mem_fwd_chan.write(instr);
176:   }
177: }
178: 
179: void writeback(ClkInstChan &in)
180: {
181:   while (true) {
182:     instr_t instr = in.get();
183:     update_pipe_report(WritebackStage,instr);
184:     arch.commit(instr);
185:   }
186: }

194: void microarch_setup()
195: {
196:   arch.set_instr_class("lb", Load);
197:   arch.set_instr_class("lbu",Load);
198:   arch.set_instr_class("ld", Load);
199:   arch.set_instr_class("lf", Load);
200:   arch.set_instr_class("lh", Load);
201:   arch.set_instr_class("lhu",Load);
202:   arch.set_instr_class("lw", Load);
203:   arch.set_instr_class("sb", Store);
204:   arch.set_instr_class("sd", Store);
205:   arch.set_instr_class("sf", Store);
206:   arch.set_instr_class("sh", Store);
207:   arch.set_instr_class("sw", Store);
208:   arch.set_instr_class("beqz", Branch);
209:   arch.set_instr_class("bfpf", Branch);
210:   arch.set_instr_class("bfpt", Branch);
211:   arch.set_instr_class("bnez", Branch);
212:   arch.set_instr_class("j",    Branch);
213:   arch.set_instr_class("jal",  Branch); 
214:   arch.set_instr_class("jalr", Branch);
215:   arch.set_instr_class("jr",   Branch);
216:   arch.set_instr_class("halt", Branch);
217:   arch.set_halt_func(hangup);
218: 
219:   set_stage_name(FetchStage,    "fetch    ");
220:   set_stage_name(DecodeStage,   "decode   ");
221:   set_stage_name(ExecStage,     "exec     ");
222:   set_stage_name(MemStage,      "mem      ");
223:   set_stage_name(WritebackStage,"writeback");
224: }

226: // Watchdog timer stops the simulation after a specified amount of time.
227: void watchdog(unsigned n) 
228: {
229:   cout << "\nTimeout is " << n << " cycles\n\n";
230:   while (true) {
231:     atclock(10);
232:     printf(".");
233:     if (pTime() > n) {
234:       cout << "\nStopping- timeout at " << pTime() << endl;
235:       pExit(1);
236:     }
237:   }
238: }

240: // This function is called when the machine is halted.
241: void hangup()
242: {
243:   if (options._ep) {
244:     cout << "\n<STATE>\n";
245:     arch.print_state(cout,true);
246:     cout << "\n";
247:     arch.print_mem(cout,true);
248:     cout << "</STATE>\n";
249:   } else {
250:     // Make sure that there's a final pipeline report.
251:     if (options._pr) {
252:       pipe_report(cout,0);
253:     }
254:     cout << "\n\n"
255:          << "End of simulation at time " << dec << pTime() << "\n"
256:          << "Number of fetches:  " << arch.get_num_fetches() << "\n"
257:          << "Number of commits:  " << arch.get_num_commits() << "\n";
258:     arch.print_state(cout);
259:     cout << "\n";
260:     arch.print_mem(cout);
261:   }
262:   pExit(0);
263: }

265: int pipeline_report(unsigned delay) 
266: {
267:   pDelay(delay);
268:   while (true) {
269:     atclock();
270:     pipe_report(cout,delay);
271:   }
272:   return 0;
273: }

281: bool init(int argc,const char *argv[])
282: {
283:   printf("\n\n\n\n"
284:          "Simple 5-stage DLX pipeline\n");
285: 
286:   if (!process_args(options,argc,argv)) {
287:     return false;
288:   }
289:   
290:   if (!load_files(arch,options._files)) {
291:     return false;
292:   }
293: 
294:   if (options._pr) {
295:     spawn(pipeline_report(1));
296:   }
297: 
298:   microarch_setup();
299: 
300:   printf("\nCode and registers initialised.");
301:   return true;

304: int pMain(int argc,const char *argv[]) 
305: {
306:   if (!init(argc,argv)) {
307:     return usage(argv[0]);
308:   }
309: 
310:   //
311:   // Define the communication channels.
312:   // 
313: 
314:   ClkInstChan 
315:     fetch2decode(Clock),
316:     br_decode2fetch(Clock),
317:     decode2exec(Clock),
318:     exec2mem(Clock),
319:     mem2wb(Clock/2);
320: 
321:   NbInstChan
322:     fetch2fetch(Clock),
323:     decode2decode(Clock),
324:     exec2decode(Clock),
325:     exec2exec(Clock),
326:     mem2decode(Clock),
327:     mem2exec(Clock),
328:     mem2mem(Clock);
329: 
330:   ClkIntChan
331:     decode2fetch(Clock);
332: 
333:   //
334:   // Define the connectivity and launch each thread.
335:   //
336:   par {
337:     watchdog(10000);
338:     fetch(fetch2fetch,br_decode2fetch,decode2fetch,fetch2decode);
339:     decode(fetch2decode,decode2fetch,br_decode2fetch,decode2decode,exec2decode,mem2decode,decode2exec);
340:     exec(decode2exec,mem2exec,exec2exec,exec2decode,exec2mem);
341:     mem(exec2mem,mem2decode,mem2exec,mem2mem,mem2wb);
342:     writeback(mem2wb);
343:   }
344: 
345:   // We never get here- the watchdog stops the simulation.
346:   return 0;
347: }
