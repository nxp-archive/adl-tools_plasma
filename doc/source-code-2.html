 14: //	this model says the eav is 
 15: //	a set of user processes who squirt messages every now and then
 16: //      to a comms multiplexer
 17: //	a comms multiplexer who passes the messages on to a free transaction
 18: //      agent (there are fewer agents than terminals) 
 19: //	tp agents who call upon database routines to effect transactions and
 20: //      who pass messages on to a memory manager 
 21: //	the memory manager plays with journals and records and so forth, waiting
 22: //      for access to complete when needed.
 23: //	the system behavior is explored by having the memory manager sometimes use
 24: //      fast memory for storage and sometimes use a disk manager; the ratio 
 25: //      between the two is a cache hit rate (ignoring writethru) and on an
 26: //      enterprise attack vehicle the hit rate is assumed to be 100%
 27: //	this version doesn't worry about contention for record locks between
 28: //      transactions
 29: //	this version uses a TP-1 like transaction - two reads and a write

149: // a user
150: void user(Chan &in, Chan &out, int32 usernum, int32 max_in,
151:           int32 &transactions1, int32 &times1, int32 &transactions, int32 &times)
152: {
153:   int32 sigma_t = 0, dt;
154:   int32 count = 0;
155:   pDelay(Rand.uniform(0, 0, ThinkTime * msec)); //stagger the start-ups of the users
156:   ptime_t now, max = max_in;
157:   while ((now = pTime()) < max) {
158:     int32 dummy, t;
159:     t = now;
160:     dprintf ("user %d:  Sent request\n",usernum);
161:     out.write(usernum);
162:     dummy = in.get();
163:     dprintf ("user %d:  Got response.\n",usernum);
164:     dt = (now - t)/msec;            // delta t in milliseconds
165:     count = count + 1;              // number of transactions
166:     sigma_t = sigma_t + dt;             // total response time
167:     transactions1 = transactions1 + 1;      // global transaction sigma
168:     times1 = times1 + dt;                   // global response time sigma
169:     pDelay(Rand.uniform(0, (ThinkTime - ThinkVar) * msec, 
170:                  (ThinkTime + ThinkVar) * msec));
171:     dprintf ("user %d:  Total transactions %d at time %lld\n",usernum,transactions1,now);
172:   }
173:   dprintf ("user %d:  Done.\n");
174:   transactions = transactions + count;
175:   times = times + sigma_t;
176: }

208: //////////////////////////// timestamp /////////////////////////////
209: 
210: // simply prints time every dt units
211: void timestamp(int32 dt, int32 maxtime, int32 &transactions, int32 &time) 
212: {
213:   int32 newtime = 0;
214:   ptime_t EndTime;
215:   int32 LastCount = 0, LastTime = 0;
216:   int32 newtrans, newdt, oldtrans = 0, olddt = 0;
217:     
218:   EndTime = (maxtime + (10 * ((ThinkTime + ThinkVar)* msec)));
219:   if (maxtime < 0) {
220:     maxtime = 0x7fffffff;
221:   }
222:   if (EndTime < 0) {
223:     EndTime = 0x7fffffff;
224:   }
225:   while(1) {
226:     if (pTime() > EndTime) {
227:       say("\n====Running much longer than expected; asked-for time is ");
228:       printfixed(maxtime, msec);
229:       say("\n\tmsec and now is ");
230:       saynum(pTime()/msec);
231:       say(" msec.\n\n");
232:       curr_stats(maxtime, transactions, time);
233:       pExit(1);
234:     }
235:     
236:     curr_stats(maxtime, transactions, time);
237:     // now give average msec/transaction since last time
238:     newtrans = transactions - LastCount;
239:     newdt   = time - LastTime;
240:     if (newtrans > 0) {     
241:       int32 n, t;   
242:       say(" (");
243:       t = (newdt + olddt)/2;
244:       n = (newtrans + oldtrans)/2;
245:       printfixed(t, n);
246:       olddt = t;
247:       oldtrans = n;
248:       LastCount = transactions;
249:       LastTime	= time; 
250:       say("ms/t)");
251:     }
252:   
253:     say("->");
254:     printfixed(newtrans, dt/sec);
255:     say("tps.");
256:     sayflush();
257:     pDelay((newtime + dt) - pTime());
258:     newtime = newtime + dt;
259:   }
260: }

262: /////////////////////////// comms_in //////////////////////////
263: 
264: // comms_in listens to the terminals and sends their messages on to
265: //	one of the available transaction processes
266: 
267: void comms_in(int32 mips, Chans &in, Chans &out, Chans &free) 
268: {
269:   int32 terminal, tp;
270:   say("\nComms starting.");
271:   while (1) {
272:     afor (int i = 0; i != (int)in.size(); ++i) {
273:       in[i].port(int v) {
274:         // Drain a value from the input channels.
275:         terminal = v;
276:         dprintf ("comms_in:  Got a value from terminal %d\n",v);
277:       }
278:     }
279:     // Got input from a terminal; now find a free tp process
280:     afor (int i = 0; i != (int)free.size(); ++i) {
281:       free[i].port() {
282:         tp = i;
283:         dprintf ("comms_in:  Got a tp process %d\n",tp);
284:       }
285:     }
286:     // Compute.
287:     think(mips, TermInPath);
288:     dprintf ("comms_in:  Finished thinking.\n");
289:     // Send value.
290:     out[tp].write(terminal);
291:     dprintf ("comms_in:  Wrote value %d to tp %d\n",terminal,tp);
292:   }
293: }

319: // these routines 'do transactions'; they call upon the memory manager
320: //	(who can be in essence a ram manager or a disk manager) to read,
321: //	lock, write, free etc things, and they wait for writes to complete
322: //	appropriately thus getting the right amount of performance-sapping
323: //	synchronisation
324: 
325: //	in this version, everybody just tosses pennies; there's no representation
326: //	of real records. also, we assume no contention for records for the moment
327: //	and further, we assume that no record is locked over multiple terminal
328: //	requests.

334: //////////////////////// disk_manager //////////////////////////
335: 
336: // the disk manager listens to all the tp agents who may sometimes
337: // call upon him to do disk accesses
338: // the disk manager runs a number of disks, which he chooses between 
339: // at random for a request. each disk is represented by a process
340: class DiskManager {
341: 
342: public:
343: 
344:   // a sequential buffer implementation
345:   void drive_buffer(int32 buff_size, Chan &in, Chan &out, int32 number) 
346:   {
347:     IntVect stuff(buff_size);
348: 
349:     Chan in_to_out, out_to_in;
350:     bool in_asleep = false, out_asleep = true;
351:     int32 inptr = 0, outptr = 0, count = 0;
352: 
353:     // the input handler    
354:     par {
355:       {
356:         int32 any;
357:         while(1) {
358:           int32 data = in.get();
359:           stuff[inptr] = data;
360:           inptr = (inptr + 1) % buff_size;	// increment in pointer
361:           count = count + 1;            // increment count
362:           if (out_asleep) {
363:             out_asleep = false;
364:             in_to_out.write(any);
365:           }
366:           if (count == buff_size) {
367:             // buffer is full
368:             in_asleep = true;
369:             any = out_to_in.get();
370:           }
371:         }
372:       }
373:       {
374:         // the output handler   
375:         int32 data, any;
376:         while(1) {
377:           // first see if we got any data
378:           if (out_asleep) {
379:             any = in_to_out.get();	// wait for stuff
380:           }
381:           data = stuff[outptr];
382:           out.write(data);
383:           outptr = (outptr + 1) % buff_size;
384:           count = count - 1;
385:           if (in_asleep) {
386:             // in has gone to sleep
387:             in_asleep = false;
388:             out_to_in.write(any);   // wake him up
389:           }
390:           if (count == 0) {
391:             out_asleep = true;
392:           }
393:         }    
394:       }
395:     }
396:   }
397: 
398:   //////////////////////// drive /////////////////////////
399: 
400:   void drive(Chan &in, Chan &out, int32 n) 
401:   {
402:     int32 cmnd;
403:     const int32 rotation_time = 10 * msec;
404:     const int32 seek_time = 100 * musec;    // per track
405:     const int32 settling_time = 1 * msec;
406:     const int32 num_tracks	= 100;
407:     // these disks have a rotation speed of 10 msec, a head movement and
408:     // settling time of 100musec/track + 1 msec, 100 tracks, and are 
409:     // unsynchronised.
410:     
411:     int32 agent;
412:     int32 track = 0;
413:     int32 newtrack = 0;
414:     int32 dtrack;
415:     
416:     pDelay(Rand.uniform(0, 10 * msec, 20 * msec));
417:     while(1) {
418:       int32 wait_time;
419:       
420:       cmnd = in.get();             // get the composite command
421:       agent = cmnd >> 16;          // resolve into agent number
422:       cmnd = cmnd & 255;           // ..and command
423:       newtrack = Rand.uniform(0, 0, num_tracks);
424:       if (newtrack < track) {
425:     dtrack = track - newtrack;
426:       } else {
427:         dtrack = newtrack - track;
428:       }
429:       wait_time = Rand.uniform(0, 0, rotation_time) + // rotate latency
430:         (dtrack * seek_time) +      // seek latency
431:         settling_time;
432:       pDelay(wait_time);
433:       // synchronous command?
434:       if ((cmnd == Read) || (cmnd == Write_synch)) {
435:     out.write(agent);
436:       }
437:       pDelay(100 * musec);
438:     }
439:   }

441:   ///////////////////////// manager /////////////////////////
442:     
443:   // Accepts commands from tp agents and sends to drives
444:   void manager(int32 mips, int32 disks, Chans &commands, Chans &to_drives)
445:   {  
446:     int32 cmnd, agent, drivenum, composite;
447:     say("\nDisk manager starting.");    
448:     while(1) {
449:       dprintf ("disk manager:  waiting for command.\n",cmnd,agent);
450:       afor(unsigned i = 0; i != commands.size(); ++i) {
451:         commands[i].port(int v) {
452:           cmnd = v;
453:           agent = i;
454:           dprintf ("disk manager:  command %d from %d.\n",cmnd,agent);
455:         }
456:       }
457:       // got a request from a tp agent; assign it to a drive
458:       drivenum = Rand.uniform(0, 0, disks);     // chooses rv in 0..disks-1
459:       composite = cmnd + (agent << 16);
460:       think(mips, DiskCmndPath);        // IO-issuing path length
461:       to_drives[drivenum].write(composite);
462:       dprintf ("disk manager:  wrote message to drives.\n");
463:     }
464:   }
465: 
466:   ///////////////////////// responder ////////////////////////////
467: 
468:   // the disk interrupt handler; catches 'done' messages from disks,
469:   // sends signal to appropriate tp agent
470:   void responder(int32 mips, Chans &to_agents, Chans &from_disks) 
471:   {
472:     int32 agent;
473:     while(1) {
474:       afor (unsigned i = 0; i != from_disks.size(); ++i) {
475:         from_disks[i].port(int v) {
476:           agent = v;
477:         }
478:       }
479:       to_agents[agent].write(agent);
480:       think(mips, DiskResponsePath);            
481:     }
482:   }
483: 
484: public:
485:   // Now, the body of the disk manager just runs the manager on the
486:   // computer and the drives on themselves
487:   void operator()(int32 mips, int32 buff_size, int32 disks, Chans &in, Chans &out) 
488:   {
489:     Chans to_drives(disks), from_drives(disks), buff_to_drives(disks);
490:     
491:     par {
492:       par {
493:         on (mainframe) {
494:           manager(mips, disks, in, to_drives);
495:         }
496:         on (mainframe) {
497:           responder(mips, out, from_drives);
498:         }
499:       }
500:       pfor (int i = 0; i != disks; ++i) {
501:         on (diskdrives[i]) {
502:           drive(buff_to_drives[i], from_drives[i], i);
503:         }
504:       }
505:       pfor (int i = 0; i != disks; ++i) {      
506:         on (diskdrives[i]) {
507:           drive_buffer(buff_size, to_drives[i], buff_to_drives[i], i); 
508:         }
509:       }
510:     }
511:   }

538: //////////////////////////// MEMORY /////////////////////////////
539: 
540: // asks for stuff from the memory system; in a normal system, this means
541: // going to a disk; in a giant memory system, it means reading memory. we
542: // treat the thing as cached, with a supplied hit rate. with 100% hit, we
543: // always go to memory; with 90% hit rate we go to disk 10% of the time
544: 
545: void mem_read(Chan &to_disks, Chan &from_disks, int32 hit_rate) 
546: {
547:   dprintf ("mem_read:  starting.\n");
548:   if (Rand.uniform(0, 0, 1000) > hit_rate) {
549:     dprintf ("mem_read:  disk write asynch start.\n");
550:     // hit_rate is an integer in 0..999; if our rv is larger than hitrate
551:     //then we have a MISS   
552:     disk_read(to_disks, from_disks);
553:     dprintf ("mem_read:  disk write asynch finish.\n");
554:   } else {
555:     // memory read; this is quick; assume 10 microseconds
556:     dprintf ("mem_read:  delay start.\n");
557:     pDelay(15 * musec);
558:     dprintf ("mem_read:  delay finish.\n");
559:   }
560:   dprintf ("mem_read:  finished.\n");
561: }
562:     
563: void mem_write_synch(Chan &to_disks, Chan &from_disks, int32 hit_rate) 
564: {
565:   // gives stuff to the memory system and waits for write to finish
566:   if (Rand.uniform(0, 0, 1000) > hit_rate) {
567:     // hit_rate is an integer in 0..999; if our rv is larger than hitrate
568:     // then we have a MISS
569:     disk_write_synch(to_disks, from_disks);
570:   } else {
571:     // memory write; this is quick; assume 10 microseconds
572:     pDelay(15 * musec);
573:   }
574: }

637:   after_journal(to_disks, from_disks, hit_rate);
638: }

640: ///////////////////////// TP /////////////////////////////
641: 
642: // the tp process does all the transaction work. there are multiple
643: //	instantiations; each catches stuff on its in, passes stuff on through
644: //	its out and signals its free to the comms mux via the free channel
645: 
646: class Agents {
647: 
648: public: 
649:   void lock_record(int32 mips) 
650:   {
651:     // we simply compute a bit -    locks are in memory
652:     think(mips, LockPath);          
653:   }
654:     
655:   void unlock_record(int32 mips) 
656:   {
657:     // same again
658:     think(mips, UnlockPath);
659:   }
660: 
661:   void tp_agent(int32 mips, Chan &in, Chan &out, Chan &free, Chan &to_disks, Chan &from_disks,
662:                 int32 tp_num, int32 hit_rate)
663:   {         
664:     int32 terminal = 0,freecount = 0;
665:     pSetPriority(3);            // lower priority
666:     while(1) {
667:       dprintf ("tp agent  %d:  is free.\n",tp_num);
668:       free.write(freecount++);	// signal i'm free
669:       dprintf ("tp agent  %d:  waiting for input.\n",tp_num);
670:       terminal = in.get();	// get some input
671:       dprintf ("tp agent %d:  got message from terminal %d.\n",tp_num,terminal);
672:       lock_record(mips);    // lock all the records
673:       lock_record(mips);
674:       lock_record(mips);
675:       dprintf ("tp agent  %d:  done with locking record.\n",tp_num);
676:             
677:       read_record(mips, to_disks, from_disks, hit_rate);
678:       read_record(mips, to_disks, from_disks, hit_rate);
679:       dprintf ("tp agent  %d:  done with reading record.\n",tp_num);
680:       update_record(mips, to_disks, from_disks, hit_rate);
681:  
682:       dprintf ("tp agent  %d:  sent message to terminal %d.\n",tp_num,terminal);
683:       out.write(terminal);      // provide results to terminal
684:             
685:       unlock_record(mips);
686:       unlock_record(mips);
687:       unlock_record(mips);
688:       dprintf ("tp agent  %d:  records unlocked.\n",tp_num);
689:     }
690:   }
691: 
692: public:
693:   void operator()(int32 mips, Chans &in, Chans &out, Chans &free, Chans &to_disks, Chans &from_disks, 
694:          int32 num_agents, int32 hit_rate) 
695:   {
696:     //	the body, forking the agents..
697:     say("\nTP agent starting.");
698:     pfor (int i = 0; i != num_agents; ++i) {
699:       tp_agent(mips, in[i], out[i], free[i], to_disks[i], from_disks[i], i, hit_rate);
700:     }
701:   }
702: 
703: };
